        let currentMode = 'budget';

        function switchMode(mode) {
            currentMode = mode;
            document.querySelectorAll('.mode-btn').forEach(btn => {
                btn.classList.toggle('active', btn.textContent.includes(mode));
            });
            
            document.getElementById('budget-mode').classList.toggle('hidden', mode !== 'budget');
            document.getElementById('rendite-mode').classList.toggle('hidden', mode !== 'rendite');
            document.getElementById('budget-result').classList.toggle('hidden', mode !== 'budget');
            document.getElementById('rendite-result').classList.toggle('hidden', mode !== 'rendite');
        }

        function formatCurrency(value) {
            return new Intl.NumberFormat('de-DE', { 
                style: 'currency', 
                currency: 'EUR',
                maximumFractionDigits: 0
            }).format(value);
        }

        function calculate() {
            const mieteIst = parseFloat(document.getElementById('miete-ist').value);
            const mieteSoll = parseFloat(document.getElementById('miete-soll').value);
            const wohnflaeche = parseFloat(document.getElementById('wohnflaeche').value);
            
            if(currentMode === 'budget') {
                const zielrendite = parseFloat(document.getElementById('zielrendite').value)/100;
                const jahresmehrertrag = (mieteSoll - mieteIst) * wohnflaeche * 12;
                const sanierungsbudget = jahresmehrertrag / zielrendite;
                const budgetProQm = sanierungsbudget / wohnflaeche;

                document.getElementById('sanierungs-budget').textContent = formatCurrency(sanierungsbudget);
                document.getElementById('budget-pro-qm').textContent = formatCurrency(budgetProQm) + '/m²';
            } else {
                const budget = parseFloat(document.getElementById('budget').value);
                const jahresmehrertrag = (mieteSoll - mieteIst) * wohnflaeche * 12;
                const rendite = (jahresmehrertrag / budget * 100).toFixed(1) + '%';

                document.getElementById('rendite-ist').textContent = rendite;
                document.getElementById('miet-steigerung').textContent = formatCurrency(jahresmehrertrag);
            }
        }

        document.getElementById('zielrendite').addEventListener('input', function() {
            document.getElementById('zielrendite-anz').textContent = this.value + '%';
            document.getElementById('zielrendite-faktor').textContent = (100/parseFloat(this.value)).toFixed(1);
            if(currentMode === 'budget') calculate();
        });

        document.querySelectorAll('input[type="number"]').forEach(input => {
            input.addEventListener('input', calculate);
        });

        calculate();
